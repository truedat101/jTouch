jTouch(prototype) ~ Sample

This is Gesture Library for MoblieSafari(iPad,iPhone,iPodTouch)
Please gesure !

How to use

jTouch.addTouchListener( element,{ 
       callBack : function(e){
         if( e.tGesture == jTouch.gesture.filp){
           alert( "filp! (" + e.fingers+ " fingers)");
         }
         else if( e.tGesture == jTouch.gesture.swipe){
           alert( "swipe! (" + e.fingers+ " fingers)");
         }
         else if( e.tGesture == jTouch.gesture.pinchIn){
           alert( "pinch(in)!");
         }
         else if( e.tGesture == jTouch.gesture.pinchOut){
           alert( "pinch(out)!");
         }
       },
       moveCallBack : function( e){
         e.preventDefault();
       },
       startCallBack : function( e){
         e.preventDefault();
       }
});
